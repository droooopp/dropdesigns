CONCEPT

ZOZO買取 キャンペーン内、簡易査定の実装を担当いたしました。
jsonで持っている性別、キーワード、ブランド名、金額から簡易的な査定を行う機能を実装しました。

(function($, window, document, undefined) {
	var __$win = $(window);

	$(function(){
		__sexTab();
		__brand();
	});



	/**
	 * ------------------------------------------------------------
	 * 性別タブクリック
	 * ------------------------------------------------------------
	 */
	function __sexTab() {
		_init();



		/* ----------------------------------------
		 * 初期化
		 * ---------------------------------------- */

		function _init() {
			_setElement();
		}



		/* ----------------------------------------
		 * 要素設定
		 * ---------------------------------------- */

		function _setElement() {
			var $tabTrigger = $('.tabItem');
			var $tabList = $('.tabListContainer');
			var $tabMen = $('.tabMen');
			var $tabWomen = $('.tabWomen');
			var $tabContentMen = $('.tab-men');
			var $tabContentWomen = $('.tab-women');
			var $checkTrgger = $('.customRadio');
			var $checkMen = $('.check-men');
			var $checkWomen = $('.check-women');

			$tabTrigger
				.click(function(){
					var $this = $(this);

					if($this.hasClass('current')) {
					}
					else {
						$tabTrigger.removeClass('current');
						$this.addClass('current');

						if($this.hasClass('tabMen')){
							$tabList.hide();
							$tabContentMen.show();
							$checkTrgger.prop('checked',false);
							$checkMen.prop('checked',true);
						}
						else if($this.hasClass('tabWomen')) {
							$tabList.hide();
							$tabContentWomen.show();
							$checkTrgger.prop('checked',false);
							$checkWomen.prop('checked',true);
						}
						__resetCategory();
						__resetBrand();
						__resetBrandModal();
					}
				});

			$checkTrgger
				.change(function(){
					var $this = $(this);

					if($this.hasClass('check-men')){
						$tabTrigger.removeClass('current');
						$tabMen.addClass('current');
						$tabList.hide();
						$tabContentMen.show();
						$checkTrgger.prop('checked',false);
						$checkMen.prop('checked',true);
					}
					else if($this.hasClass('check-women')) {
						$tabTrigger.removeClass('current');
						$tabWomen.addClass('current');
						$tabList.hide();
						$tabContentWomen.show();
						$checkTrgger.prop('checked',false);
						$checkWomen.prop('checked',true);
					}
					__resetCategory();
					__resetBrand();
					__resetBrandModal();
				});
		}
	}

	 * ------------------------------------------------------------
	 * ブランド

	function __brand() {
		_init();



		/* ----------------------------------------
		 * 初期化
		 * ---------------------------------------- */

		function _init() {
			_getBrand();
		}



		/* ----------------------------------------
		 * ブランド取得
		 * ---------------------------------------- */

		function _getBrand() {
			var $trigger = $('.brandModalItem');
			var $sexCheck;
			var sex;
			var brandKey;
			var brandListSex;
			var brandListKey;
			var brandListName;
			var $priceModalInner = $('#priceModal-brandContainer');
			var count;

			$trigger
				.click(function() {
					var $this = $(this);
					$sexCheck = $('[name="sexCheck"]:checked');
					brandKey = $this.attr('data-brandkey');
					sex = $sexCheck.val();
					var brandListSrc = '<div class="indexTable indexTableBottomBordered m-y-0"><ul class="indexList">';
					count = 0;

					//ボタンのカレント
					$trigger.removeClass('brandModalItem-selected');
					$this.addClass('brandModalItem-selected');

					//モダール内を空に
					$priceModalInner.empty();

					for(var i=0, max=brandlist.length; i<max; i++){
						brandListSex = brandlist[i].sex;
						brandListKey = brandlist[i].keyword;
						brandListBrand = brandlist[i].brand;

						//アルファベットが一致した場合
						if(brandListKey == brandKey) {
							//性別がどちらも含む場合
							if($.inArray(sex, brandListSex) == 0 || $.inArray(sex, brandListSex) == 1){
								brandListSrc += '<li class="indexItem"><div class="indexItemAction directionRight"><div class="indexItemContent"><div class="indexItemH">'+brandListBrand+'</div></div></div></li>';
								count = count+1;
							}
							//性別が一致した場合
							if(brandListSex == sex){
								brandListSrc += '<li class="indexItem"><div class="indexItemAction directionRight"><div class="indexItemContent"><div class="indexItemH">'+brandListBrand+'</div></div></div></li>';
								count = count+1;
							}
						}
					}
					brandListSrc += '</ul></div>';

					//見出しのアルファベット変換
					brandKey = brandKey.toUpperCase();
					if(brandKey == 'OTHER'){
						brandKey = 'その他';
					}
					//見出しを追加
					$priceModalInner.append('<div class="subsectionHeader" data-dismiss="modal">'+
							'	<h3>'+
							'	'+brandKey+'('+count+'件)'+
							'	</h3>'+
							'</div>'
							);

					//ボタンを追加
					$priceModalInner.append(brandListSrc);

					//ブランドがない場合

					if(count == 0){
						$priceModalInner.append('<p class="textNote textCenter m-y-lg">ブランドがありません。</p>');
					}

					_selectBrand();

					//モーダル初期化
					var $brandModalItem = $('.brandModalItem');
					var $modalClose = $('[data-dismiss="modal"]');
					$modalClose.click(function(){
						$brandModalItem.removeClass('brandModalItem-selected');
						$priceModalInner.empty();
					});

				});
		}


		/* ----------------------------------------
		 * ブランド決定
		 * ---------------------------------------- */

		function _selectBrand() {
			var $brandItem = $('#priceModal-brandContainer .indexItem');
			var $modalClose = $('[data-dismiss="modal"]');
			var brandTxt;
			var $brandTxtWrap = $('#brandTxt');

			$brandItem
				.on('click',function(){
					$this = $(this);
					brandTxt = $this.find('.indexItemH').text();
					$brandTxtWrap.text(brandTxt)
					$modalClose.click();

					//カテゴリーアクティブ化
					var $categoryOpen = $('#categoryOpen');
					$categoryOpen.prop('disabled', false);

					//価格表示エリア
					var $priceResult = $('#priceResult');
					$priceResult.removeClass('priceAssessResultTxt-selected').text('カテゴリーを選択してください');

					__resetBrandModal();
				});

		}
	}



	/**
	 * ------------------------------------------------------------
	 * ブランドモーダル初期化

	function __resetBrandModal() {

		_init();



		/* ----------------------------------------
		 * 初期化
		 * ---------------------------------------- */

		function _init() {
			_setElement();
		}



		/* ----------------------------------------
		 * 要素設定
		 * ---------------------------------------- */

			var $priceModalInner = $('#priceModal-brandContainer');
			var $brandModalItem = $('.brandModalItem');

			//モダール内を空に
			$priceModalInner.empty();
			//アルファベットのボタンをリセット
			$brandModalItem.removeClass('brandModalItem-selected');
		}
	}



	/**
	 * ------------------------------------------------------------
	 * ブランド初期化
	 * ------------------------------------------------------------

	function __resetBrand() {

		_init();



		/* ----------------------------------------
		 * 初期化
		 * ---------------------------------------- */

			_setElement();
		}

		 * 要素設定

			$brandTxt.text('選択');

			//価格表示エリア
			var $priceResult = $('#priceResult');
			$priceResult.removeClass('priceAssessResultTxt-selected').text('ブランド・カテゴリーを選択してください');
		}
	}



		/**
		 * ------------------------------------------------------------
		 * カテゴリー初期化
		 * ------------------------------------------------------------
		 */
		function __resetCategory() {
			_init();



			/* ----------------------------------------
			 * 初期化
			 * ---------------------------------------- */

			function _init() {
				_setElement();
			}



			/* ----------------------------------------
			 * 要素設定
			 * ---------------------------------------- */

			function _setElement() {
				var $categoryTxt = $('#categoryTxt');
				var $categoryOpen = $('#categoryOpen');

				$categoryOpen.prop('disabled', true)
				$categoryTxt.text('選択');

				//価格表示エリア
				var $priceResult = $('#priceResult');
				$priceResult.removeClass('priceAssessResultTxt-selected').text('カテゴリーを選択してください');
			}
		}
})(jQuery, window, document);
